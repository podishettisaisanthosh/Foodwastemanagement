function acceptstd(){

var ids=document.getElementById('identitymod').value;

firebase.database().ref('logindetails').child("restaurent").child(ids).remove();
alert("request accepted")

firebase.database().ref('logindetails').child("accepted").push().set(
           {
accepted:ids
       }
       );
}
function declinestd(){
  var ids=document.getElementById('identitymod').value;
  firebase.database().ref('logindetails').child("restaurent").child(ids).remove();
  alert("request declined")
  firebase.database().ref('logindetails').child("declined").push().set(
           {
declined:ids
       }
       );
}
